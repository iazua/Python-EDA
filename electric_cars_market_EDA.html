from shapely.geometry import Point

# Path to extracted file
shapefile_path = r"C:\Users\ignac\Downloads\ne_110m_admin_0_countries\ne_110m_admin_0_countries.shp"

# Load the world map from the downloaded file
world = gpd.read_file(shapefile_path)

# Inspect the columns of the world GeoDataFrame
print(world.columns)

# Filter for North America using the correct column name
north_america = world[world['REGION_UN'] == 'Americas']

# Filter out rows with invalid 'Vehicle Location' values
df_valid_location = df[df['Vehicle Location'].apply(lambda x: isinstance(x, str))]

# Create a GeoDataFrame with vehicle locations
geometry = [Point(float(x.split(' ')[1][1:-1]), float(x.split(' ')[2][:-1])) for x in df_valid_location['Vehicle Location']]
gdf = gpd.GeoDataFrame(df_valid_location, geometry=geometry)

# Plot the North American region
fig, ax = plt.subplots(figsize=(10, 7))
north_america.plot(ax=ax, color='lightgray', edgecolor='black')

# Plot vehicle locations
gdf.plot(ax=ax, marker='o', color='red', markersize=5)

# Set zoom-in limits (approximate bounding box for North America)
ax.set_xlim(-170, -50)  # Longitude range
ax.set_ylim(5, 85)      # Latitude range

plt.title('Geospatial Distribution of Electric Vehicles in North America')
plt.show()
